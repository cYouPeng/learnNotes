
create table hr_admin(
 loginid varchar2(16) not null,
 loginpwd char(32) not null,
 name nvarchar2(20) not null,
 constraint pk_hr_admin primary key(loginid)
);

create table hr_dept(
 id number(4) not null,
 name nvarchar2(20) not null,
 quickKey varchar2(20) not null,
 loc nvarchar2(50) not null,
 constraint pk_hr_dept primary key(id)
);

create table hr_job(
 id number(4) not null,
 name nvarchar2(20) not null,
 quickKey varchar2(20) not null,
 maxmoney number(9,2) not null,
 minmoney number(9,2) not null,
 constraint pk_hr_job primary key(id)
);

create table hr_staff(
 id number(4)not null,
 name nvarchar2(10) not null,
 quickkey varchar2(50) not null,
 deptid number(4) not null,
 jobid number(4) not null,
 hiredate date not null,
 money number(9,2) not null,
 email varchar2(100) not null,
 tel varchar2(20) not null,
 constraint pk_hr_staff primary key(id),
 constraint fk_hr_staff_job foreign key(jobid) references hr_job(id),
 constraint fk_hr_staff_dept foreign key(deptid) references hr_dept(id)
);

create sequence seq_hr_dept
start with 1000 increment by 10;

create sequence seq_hr_job
start with 1000 increment by 10;

create sequence seq_hr_staff
start with 1000 increment by 1;
